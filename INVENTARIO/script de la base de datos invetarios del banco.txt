create database bdinventario;
use bdinventario;

CREATE TABLE `agencia` (
  `cod_agencia` int(11) NOT NULL AUTO_INCREMENT,
  `encargado` varchar(50) DEFAULT NULL,
  `telefono` varchar(15) DEFAULT NULL,
  `direccion` varchar(100) DEFAULT NULL,
  `cod_departamento` int(11) DEFAULT NULL,
  `id_municipio` int(11) DEFAULT NULL,
  PRIMARY KEY (`cod_agencia`),
  KEY `id_departamento` (`cod_departamento`),
  KEY `id_municipio` (`id_municipio`)
) ENGINE=MyISAM DEFAULT CHARSET=latin1;

CREATE TABLE `asignaciones` (
  `id_asignaciones` int(3) NOT NULL,
  `id_empleado` int(3) DEFAULT NULL,
  `fecha_asig` char(50) DEFAULT NULL,
  `id_cargo` int(3) DEFAULT NULL,
  PRIMARY KEY (`id_asignaciones`),
  KEY `id_empleado` (`id_empleado`),
  KEY `id_cargo` (`id_cargo`)
) ENGINE=MyISAM DEFAULT CHARSET=latin1;

CREATE TABLE `cargos` (
  `id_cargo` int(20) NOT NULL,
  `cargo` char(60) DEFAULT NULL,
  PRIMARY KEY (`id_cargo`)
) ENGINE=MyISAM DEFAULT CHARSET=latin1;

CREATE TABLE `categoria` (
  `id_categoria` int(11) NOT NULL AUTO_INCREMENT,
  `nombre` varchar(50) DEFAULT NULL,
  `descripcion` varchar(100) DEFAULT NULL,
  PRIMARY KEY (`id_categoria`)
) ENGINE=MyISAM DEFAULT CHARSET=latin1;

CREATE TABLE `departamentos` (
  `cod_departamento` int(11) NOT NULL,
  `departamentos` varchar(25) DEFAULT NULL,
  PRIMARY KEY (`cod_departamento`)
) ENGINE=MyISAM DEFAULT CHARSET=latin1;

CREATE TABLE `despacho` (
  `id_despacho` int(11) NOT NULL AUTO_INCREMENT,
  `cod_agencia` int(11) DEFAULT NULL,
  `id_pedido` int(11) DEFAULT NULL,
  PRIMARY KEY (`id_despacho`),
  KEY `cod_agencia` (`cod_agencia`),
  KEY `id_pedido` (`id_pedido`)
) ENGINE=MyISAM DEFAULT CHARSET=latin1;

CREATE TABLE `detalle_devoluciones` (
  `id_detalledev` int(11) NOT NULL AUTO_INCREMENT,
  `id_activo` int(11) DEFAULT NULL,
  `precio_total` double DEFAULT NULL,
  `cantidad_activos` int(11) DEFAULT NULL,
  `id_devolucion` int(11) DEFAULT NULL,
  PRIMARY KEY (`id_detalledev`),
  KEY `id_devolucion` (`id_devolucion`),
  KEY `id_activo` (`id_activo`)
) ENGINE=MyISAM DEFAULT CHARSET=latin1;

CREATE TABLE `detalle_factura` (
  `id_detallefactura` int(11) NOT NULL,
  `id_producto` varchar(10) DEFAULT NULL,
  `producto` varchar(50) DEFAULT NULL,
  `precio_unit` double DEFAULT NULL,
  `cantidad` varchar(25) DEFAULT NULL,
  `id_factura` int(11) DEFAULT NULL,
  PRIMARY KEY (`id_detallefactura`),
  KEY `id_factura` (`id_factura`)
) ENGINE=MyISAM DEFAULT CHARSET=latin1;

CREATE TABLE `detalle_pedido` (
  `id_detalle` int(11) NOT NULL AUTO_INCREMENT,
  `id_activo` int(11) DEFAULT NULL,
  `cantidad` int(11) DEFAULT NULL,
  `total` double DEFAULT NULL,
  `id_pedido` int(11) DEFAULT NULL,
  PRIMARY KEY (`id_detalle`),
  KEY `id_activo` (`id_activo`)
) ENGINE=MyISAM DEFAULT CHARSET=latin1;

CREATE TABLE `devoluciones` (
  `id_devolucion` int(11) NOT NULL AUTO_INCREMENT,
  `id_activo` int(11) DEFAULT NULL,
  `cantidad` int(11) DEFAULT NULL,
  `ven_cargo` varchar(50) DEFAULT NULL,
  `total` double DEFAULT NULL,
  PRIMARY KEY (`id_devolucion`),
  KEY `id_activo` (`id_activo`)
) ENGINE=MyISAM DEFAULT CHARSET=latin1;


CREATE TABLE `empleado` (
  `id_empleado` int(11) NOT NULL AUTO_INCREMENT,
  `nombres` varchar(50) DEFAULT NULL,
  `apellidos` varchar(50) DEFAULT NULL,
  `direccion` varchar(100) DEFAULT NULL,
  `cod_departamento` int(11) DEFAULT NULL,
  `id_municipio` int(11) DEFAULT NULL,
  `dui` varchar(15) DEFAULT NULL,
  `nit` varchar(15) DEFAULT NULL,
  `telefono` varchar(20) DEFAULT NULL,
  PRIMARY KEY (`id_empleado`),
  KEY `cod_departamento` (`cod_departamento`),
  KEY `id_municipio` (`id_municipio`)
) ENGINE=MyISAM DEFAULT CHARSET=latin1;


CREATE TABLE `factura` (
  `id_factura` int(11) NOT NULL AUTO_INCREMENT,
  `num_factura` varchar(5) DEFAULT NULL,
  `id_proveedor` int(11) DEFAULT NULL,
  `tipo_factura` varchar(50) DEFAULT NULL,
  `fecha` varchar(30) DEFAULT NULL,
  `total` double DEFAULT NULL,
  PRIMARY KEY (`id_factura`),
  KEY `id_proveedor` (`id_proveedor`)
) ENGINE=MyISAM DEFAULT CHARSET=latin1;


CREATE TABLE `garantias` (
  `id_garantia` int(11) NOT NULL AUTO_INCREMENT,
  `num_serie` varchar(50) DEFAULT NULL,
  `tiemp_garantia` varchar(50) DEFAULT NULL,
  `id_proveedor` int(11) DEFAULT NULL,
  PRIMARY KEY (`id_garantia`),
  KEY `id_proveedor` (`id_proveedor`)
) ENGINE=MyISAM DEFAULT CHARSET=latin1;


CREATE TABLE `municipios` (
  `id_municipio` int(11) NOT NULL AUTO_INCREMENT,
  `municipio` varchar(50) DEFAULT NULL,
  `cod_departamento` int(11) NOT NULL,
  PRIMARY KEY (`id_municipio`),
  KEY `cod_departamento` (`cod_departamento`)
) ENGINE=MyISAM AUTO_INCREMENT=264 DEFAULT CHARSET=latin1;

CREATE TABLE `pedidos` (
  `id_pedido` int(11) NOT NULL AUTO_INCREMENT,
  `cod_agencia` int(11) DEFAULT NULL,
  `fecha_pedido` varchar(50) DEFAULT NULL,
  `fecha_entrega` varchar(50) DEFAULT NULL,
  `fecha_envio` varchar(50) DEFAULT NULL,
  `id_transporte` varchar(50) DEFAULT NULL,
  PRIMARY KEY (`id_pedido`),
  KEY `id_transporte` (`id_transporte`)
) ENGINE=MyISAM DEFAULT CHARSET=latin1;

CREATE TABLE `productoactivos` (
  `id_activo` int(11) NOT NULL AUTO_INCREMENT,
  `nombre` varchar(50) DEFAULT NULL,
  `descripcion` varchar(100) DEFAULT NULL,
  `estado_prod` varchar(25) DEFAULT NULL,
  `id_factura` int(11) DEFAULT NULL,
  `id_categoria` int(11) DEFAULT NULL,
  `cod_agencia` int(11) DEFAULT NULL,
  `id_pedido` int(11) DEFAULT NULL,
  PRIMARY KEY (`id_activo`),
  KEY `id_factura` (`id_factura`),
  KEY `id_categoria` (`id_categoria`),
  KEY `cod_agencia` (`cod_agencia`),
  KEY `id_pedido` (`id_pedido`)
) ENGINE=MyISAM DEFAULT CHARSET=latin1;


CREATE TABLE `proveedores` (
  `id_proveedor` int(11) NOT NULL AUTO_INCREMENT,
  `dui_pro` varchar(15) DEFAULT NULL,
  `nit_pro` varchar(15) DEFAULT NULL,
  `nom_compa` varchar(50) DEFAULT NULL,
  `nombre_contac` varchar(50) DEFAULT NULL,
  `cargo_contac` varchar(50) DEFAULT NULL,
  `direccion` varchar(10) DEFAULT NULL,
  `cod_departamento` int(11) DEFAULT NULL,
  `id_municipio` int(11) DEFAULT NULL,
  `telefono` varchar(20) DEFAULT NULL,
  `fax` varchar(20) DEFAULT NULL,
  `e_mail` varchar(50) DEFAULT NULL,
  PRIMARY KEY (`id_proveedor`),
  KEY `cod_departamento` (`cod_departamento`),
  KEY `id_municipio` (`id_municipio`)
) ENGINE=MyISAM DEFAULT CHARSET=latin1;

CREATE TABLE `transporte` (
  `id_transporte` int(11) NOT NULL AUTO_INCREMENT,
  `num_placa` varchar(15) DEFAULT NULL,
  `id_empleado` int(11) DEFAULT NULL,
  PRIMARY KEY (`id_transporte`),
  KEY `id_empleado` (`id_empleado`)
) ENGINE=MyISAM DEFAULT CHARSET=latin1;
